import matplotlib.pyplot as plt
import pandas as pd
import numpy as np

# 1. إدخال بيانات المريض (كلهم طبيعي ماعدا النبض 130)
patient_vitals = [16, 98, 0, 120, 130, 36.8, 0, 0]
labels = ['Resp', 'Oxygen', 'BP_Type', 'BP_Sys', 'Pulse', 'Temp', 'Heart', 'Risk']

# تحويل البيانات لـ DataFrame للتوقع
patient_df = pd.DataFrame([patient_vitals], columns=X.columns)
prediction = model.predict(patient_df)

# 2. إعداد بيانات مخطط الرادار (Radar Chart)
stats = np.concatenate((patient_vitals, [patient_vitals[0]]))
num_vars = len(labels)
angles = np.linspace(0, 2 * np.pi, num_vars, endpoint=False).tolist()
angles += angles[:1] # إغلاق الدائرة

# 3. رسم مخطط الرادار (الطريقة الصحيحة)
fig, ax = plt.subplots(figsize=(8, 8), subplot_kw=dict(polar=True))

# رسم الإطار والملء
ax.fill(angles, stats, color='#32CD32', alpha=0.3)
ax.plot(angles, stats, color='#228B22', linewidth=2)

# إضافة التسميات والقيم حول الدائرة
for angle, label, value in zip(angles[:-1], labels, patient_vitals):
    ax.text(angle, value + 10, f'{label}: {value}', ha='center', va='center', fontsize=10, fontweight='bold')

# تخصيص العنوان
title = f'Vital Signs Profile - Predicted Level: {prediction[0]}'
ax.set_title(title, size=15, color='#228B22', fontweight='bold', position=(0.5, 1.1))

plt.show()
